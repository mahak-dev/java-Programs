import java.sql.*;
import java.io.*;

public class MYSQLJDBCINSERT
{
public static void main(String ag[]) throws Exception
{

String sql ="INSERT INTO EMP VALUES ('MAHAK GUPTA',3,200000.00,7/02/2022,7)";

}


}
